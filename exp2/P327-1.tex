\subsection{P327-1}
在区间$[0, 1]$上用Euler法解下列初值问题，取步长$h=0.1$，保留到小数点后位。

\begin{enumerate}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\item
\begin{equation*}
\begin{cases}
y' & = \sin x + e ^ {-x} \\
y(0) & = 0
\end{cases}
\end{equation*}

\begin{SOLVE}
在MATLAB命令窗口执行
\begin{lstlisting}
>> [x, y] = m_euler(@(x, y)sin(x)+exp(-x), [0,1], 0, 0.1)

x =

         0    0.1000    0.2000    0.3000    0.4000    0.5000    0.6000    0.7000    0.8000    0.9000    1.0000


y =

         0    0.1000    0.2005    0.3022    0.4058    0.5118    0.6204    0.7318    0.8458    0.9625    1.0815

\end{lstlisting}
\end{SOLVE}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\item
\begin{equation*}
\begin{cases}
y' & = -y \\
y(0) & = 2
\end{cases}
\end{equation*}

\begin{SOLVE}
在MATLAB命令窗口执行
\begin{lstlisting}
>> [x, y] = m_euler(@(x, y) -y, [0,1], 2, 0.1)

x =

         0    0.1000    0.2000    0.3000    0.4000    0.5000    0.6000    0.7000    0.8000    0.9000    1.0000


y =

    2.0000    1.8000    1.6200    1.4580    1.3122    1.1810    1.0629    0.9566    0.8609    0.7748    0.6974

\end{lstlisting}
\end{SOLVE}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\end{enumerate}
